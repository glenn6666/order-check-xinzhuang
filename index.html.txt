<div id="loading">讀取中...</div>
<div id="results"></div>

<script>
    // 你的試算表設定
    const sheetId = '1pjTOojFm4-_rGpGGN9Nv31GCdYMPtSDQkfn8vbyBFLQ';
    const gid = '171915529';
    // CSV 導出連結
    const csvUrl = `https://docs.google.com/spreadsheets/d/${sheetId}/export?format=csv&gid=${gid}`;

    async function searchData() {
        const query = document.getElementById('nameInput').value.trim();
        if (!query) return alert('請輸入暱稱！');

        const resultDiv = document.getElementById('results');
        const loading = document.getElementById('loading');
        
        resultDiv.innerHTML = '';
        loading.style.display = 'block';

        try {
            const response = await fetch(csvUrl);
            const data = await response.text();
            const rows = data.split('\n').map(row => row.split(','));

            // 假設欄位順序：狀態(0), 備註(1), 客戶暱稱(2), 商品名稱(3), 數量(4)...
            // 根據你的圖片，客戶暱稱在第 C 欄 (Index 2)，商品在 D 欄 (Index 3)
            // 狀態在 A 欄 (Index 0)
            
            let found = false;
            // 從第 2 行開始跑（跳過標題）
            for (let i = 1; i < rows.length; i++) {
                const status = rows[i][0] ? rows[i][0].trim() : "";
                const customer = rows[i][2] ? rows[i][2].trim() : "";
                const itemName = rows[i][3] ? rows[i][3].trim() : "";
                const count = rows[i][4] ? rows[i][4].trim() : "";

                // 判斷邏輯：名字相符 且 (狀態不是已勾選 或 你指定的未取貨關鍵字)
                // 注意：CSV 導出時，勾選框通常會變成 TRUE/FALSE
                if (customer.includes(query) && status !== "TRUE") {
                    const div = document.createElement('div');
                    div.className = 'result-item';
                    div.innerHTML = `
                        <div>客戶：${customer}</div>
                        <div class="item-name">商品：${itemName} x ${count}</div>
                        <div class="status-tag">狀態：尚未取貨</div>
                    `;
                    resultDiv.appendChild(div);
                    found = true;
                }
            }

            if (!found) {
                resultDiv.innerHTML = '<p style="text-align:center; color:#888;">查無待取貨商品，或名字輸入不完全。</p>';
            }

        } catch (error) {
            console.error(error);
            alert('讀取資料失敗，請檢查網路或試算表權限。');
        } finally {
            loading.style.display = 'none';
        }
    }
</script>
